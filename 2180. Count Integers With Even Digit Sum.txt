class Solution {
    public int countEven(int num) {

        int count = 0;
        for(int i=2; i<=num; i++){
            int sum = 0;
            int a = i;

            while(a != 0){
                int lastDigit = a%10;
                sum += lastDigit;
                a = a/10;
            }

            if(sum%2 == 0){
                count++;
            }
        }
        return count;
    }
}
