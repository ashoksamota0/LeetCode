class Solution {
    public int numOfWays(int n) {
        long mod = 1000000007;
        long a = 6;
        long b = 6;
        for (int i = 2; i <= n; i++) {
            long na = (a * 3 + b * 2) % mod;
            long nb = (a * 2 + b * 2) % mod;
            a = na;
            b = nb;
        }
        return (int)((a + b) % mod);
    }
}
