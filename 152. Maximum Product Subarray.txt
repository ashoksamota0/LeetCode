//Java code
class Solution {
    public int maxProduct(int[] nums) {
        int max = nums[0], min = nums[0], ans = nums[0];

        for (int i = 1; i < nums.length; i++) {
            int n = nums[i];

            if (n < 0) {
                int temp = max;
                max = min;
                min = temp;
            }

            max = Math.max(n, max * n);
            min = Math.min(n, min * n);

            ans = Math.max(ans, max);
        }

        return ans;
    }
}
