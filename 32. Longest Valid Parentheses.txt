class Solution {
    public int longestValidParentheses(String s) {
        int max = 0;
        int[] dp = new int[s.length()];

        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) == ')') {
                int j = i - dp[i - 1] - 1;
                if (j >= 0 && s.charAt(j) == '(') {
                    dp[i] = dp[i - 1] + 2;
                    if (j - 1 >= 0) {
                        dp[i] += dp[j - 1];
                    }
                }
                max = Math.max(max, dp[i]);
            }
        }

        return max;
    }
}
