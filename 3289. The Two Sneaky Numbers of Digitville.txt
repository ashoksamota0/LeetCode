class Solution {
    public int[] getSneakyNumbers(int[] nums) {
        int[] nums1 = new int[nums.length];
        int l = -1;

        for(int i=0; i<nums.length; i++){
            for(int j=i+1; j<nums.length; j++){
                if(nums[j] == nums[i]){
                    l++;
                    nums1[l] = nums[i];
                }
            }
        }
        return Arrays.copyOf(nums1, l + 1);
    }
}
