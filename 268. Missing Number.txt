class Solution {
    public int missingNumber(int[] nums) {
        int n = nums.length;
        // System.out.println(n);

        int sum = (n*(n+1))/2;
        // System.out.println(sum);

        int sum1 = 0;
        for(int i=0; i<nums.length; i++){
            sum1 += nums[i];
        }
        // System.out.println(sum1);

        int result = sum-sum1;
        // System.out.println(result);
        return result;
    }
}
