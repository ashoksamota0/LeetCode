class Solution {
    public int maxProfit(int[] prices) {
        int minPrice = prices[0], maxProfit=0;
        for(int i=0; i<prices.length; i++){
            if(minPrice > prices[i]){
                minPrice = prices[i];
            }

            maxProfit = Math.max(maxProfit, prices[i] - minPrice);
        }
        return maxProfit;
    }
}
