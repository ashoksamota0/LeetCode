//Java code
class Solution {
    public int[] leftRightDifference(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];

        int leftSum = 0;
        int total = 0;

        for (int num : nums) total += num;

        for (int i = 0; i < n; i++) {
            total -= nums[i];
            ans[i] = Math.abs(leftSum - total);
            leftSum += nums[i];
        }

        return ans;
    }
}
