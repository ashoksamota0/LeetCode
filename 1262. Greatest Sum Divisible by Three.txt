class Solution {
    public int maxSumDivThree(int[] nums) {
        int[] dp = new int[]{0, Integer.MIN_VALUE, Integer.MIN_VALUE};
        for(int n : nums){
            int[] temp = dp.clone();
            for(int r = 0; r < 3; r++){
                int nr = (r + n) % 3;
                temp[nr] = Math.max(temp[nr], dp[r] + n);
            }
            dp = temp;
        }
        return dp[0];
    }
}
