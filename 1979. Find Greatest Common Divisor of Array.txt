class Solution {
    public int findGCD(int[] nums) {
        int min = Integer.MAX_VALUE;
        int max = Integer.MIN_VALUE;

        for(int i=0; i<nums.length; i++){
            if(min > nums[i]){
                min = nums[i]; 
            }
            if(max < nums[i]){
                max = nums[i];
            }
        }

        while(min != 0){
            int temp = max % min;
            max = min;
            min = temp;
        }

        return max;
    }
}

//Method:- All we need to know is that 1.Finding max element & min element of an array. 2.Finding the GCD of 2 numbers.
