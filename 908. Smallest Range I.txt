class Solution {
    public int smallestRangeI(int[] A, int K) {
        if (A == null || A.length == 0) return 0;
        int min = Integer.MAX_VALUE, max = Integer.MIN_VALUE;
        for (int v : A) {
            if (v < min) min = v;
            if (v > max) max = v;
        }
        int diff = max - min - 2 * K;
        return diff > 0 ? diff : 0;
    }
}
