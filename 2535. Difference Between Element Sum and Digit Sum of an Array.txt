class Solution {
    public int differenceOfSum(int[] nums) {
        int sum = 0;
        int digitSum = 0;

        for(int i=0; i<nums.length; i++){
            sum += nums[i];

            int a = nums[i];

            while(a != 0){
                int lastDigit = a%10;
                digitSum += lastDigit;
                a = a/10;
            }
        }

        int difference = Math.abs(sum - digitSum);
        return difference;
    }
}
