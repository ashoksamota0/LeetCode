class Solution {
    public long maxMatrixSum(int[][] matrix) {
        long sum = 0;
        int min = Integer.MAX_VALUE;
        int neg = 0;
        for (int[] row : matrix) {
            for (int v : row) {
                if (v < 0) neg++;
                int a = Math.abs(v);
                sum += a;
                min = Math.min(min, a);
            }
        }
        if ((neg & 1) == 1) sum -= 2L * min;
        return sum;
    }
}
