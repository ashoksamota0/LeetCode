import java.util.*;

class Solution {
    public TreeNode balanceBST(TreeNode root) {
        List<Integer> arr = new ArrayList<>();
        inorder(root, arr);
        return build(arr, 0, arr.size() - 1);
    }

    void inorder(TreeNode node, List<Integer> arr) {
        if (node == null) return;
        inorder(node.left, arr);
        arr.add(node.val);
        inorder(node.right, arr);
    }

    TreeNode build(List<Integer> arr, int l, int r) {
        if (l > r) return null;
        int m = (l + r) / 2;
        TreeNode root = new TreeNode(arr.get(m));
        root.left = build(arr, l, m - 1);
        root.right = build(arr, m + 1, r);
        return root;
    }
}
