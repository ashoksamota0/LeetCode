class Solution {
    public int commonFactors(int a, int b) {
        
        while(b != 0){
            int temp = a%b;
            a = b;
            b = temp;
        }

        int count = 0;
        for(int i=1; i<=a; i++){
            if(a%i == 0){
                count++;
            }
        }

        return count;
    }
}
