class Solution {
    public int maximumProduct(int[] nums) {
        
        Arrays.sort(nums);
        int n = nums.length;
        int method1 = nums[0] * nums[1] * nums[n-1];
        int method2 = nums[n-1] * nums[n-2] * nums[n-3];

        //max(max1∗max2∗max3,min1∗min2∗max1)
        return Math.max(method2, method1);
    }
}
