//Java code
class Solution {
    public int maxRotateFunction(int[] nums) {
        long sum = 0, f = 0;
        int n = nums.length;

        for (int i = 0; i < n; i++) {
            sum += nums[i];
            f += (long) i * nums[i];
        }

        long max = f;

        for (int k = 1; k < n; k++) {
            f = f + sum - (long) n * nums[n - k];
            max = Math.max(max, f);
        }

        return (int) max;
    }
}
