class Solution {
    public int alternateDigitSum(int n) {
        int result = 0;
        int reversed = 0;

        int temp = n;
        while(temp != 0){
            int lastDigit = temp%10;
            reversed = reversed*10 + lastDigit;
            temp = temp/10; 
        }

        int count = 0;
        while(reversed != 0){
            count++;
            int lastDigit = reversed%10;

            if(count%2 != 0){
                result += lastDigit;
            }
            else{
                result -= lastDigit;
            }
        
            reversed = reversed/10;
        }

        return result;
    }
}
