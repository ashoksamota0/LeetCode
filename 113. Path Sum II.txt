import java.util.*;

class Solution {
    public List<List<Integer>> pathSum(TreeNode root, int targetSum) {
        List<List<Integer>> res = new ArrayList<>();
        backtrack(root, targetSum, new ArrayList<>(), res);
        return res;
    }

    void backtrack(TreeNode node, int sum, List<Integer> cur, List<List<Integer>> res) {
        if (node == null) return;

        cur.add(node.val);

        if (node.left == null && node.right == null && sum == node.val) {
            res.add(new ArrayList<>(cur));
        } else {
            backtrack(node.left, sum - node.val, cur, res);
            backtrack(node.right, sum - node.val, cur, res);
        }

        cur.remove(cur.size() - 1);
    }
}
